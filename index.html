// --- LOGIQUE DE RECHERCHE AUTOMATISÉE --- function searchFDS(code, name) { const query = name ? `${name} ${code}` : code; // On propose 2 sources solides : // 1. SDS Manager (Excellent pour le scan direct) const sdsUrl = `https://www.sdsmanager.com/fr/search?q=${encodeURIComponent(query)}`; // 2. Recherche ciblée Google (Force le PDF et le domaine FDS) const googleUrl = `https://www.google.com/search?q=filetype:pdf+FDS+${encodeURIComponent(query)}`; // On ouvre SDS Manager par défaut car c'est le plus "pro" pour les codes-barres window.open(sdsUrl, '_blank'); // On affiche aussi le lien Google dans l'interface au cas où return googleUrl; } // --- MODIFICATION DE LA SAUVEGARDE --- function saveProduct() { const name = document.getElementById('prod-name').value; const code = document.getElementById('prod-code').value; const fileInput = document.getElementById('prod-fds'); if(!name && !code) return alert("Veuillez scanner un code ou entrer un nom."); // Si pas de fichier uploadé, on génère un lien de recherche auto let fdsLink = ""; if (fileInput.files.length === 0) { fdsLink = searchFDS(code, name); } const reader = new FileReader(); reader.onload = function(e) { finalizeSave(name, code, e.target.result); }; if (fileInput.files.length > 0) { reader.readAsDataURL(fileInput.files[0]); } else { finalizeSave(name, code, fdsLink); } } function finalizeSave(name, code, fds) { const newProduct = { id: Date.now(), name: name || "Produit Inconnu", code: code, fds: fds, // Soit le Base64 du fichier, soit le lien de recherche Google date: new Date().toLocaleDateString() }; inventory.push(newProduct); localStorage.setItem('restev_inventory', JSON.stringify(inventory)); alert("Produit enregistré !"); nav('page-list', 'nav-list'); }
